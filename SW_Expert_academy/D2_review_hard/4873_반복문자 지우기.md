```py
T = int(input())
for stage in range(1, T+1):
    chs = list(input())
    result = ['None',chs[0]]
    pre = chs[0]
    for i in range(1, len(chs)):
        if pre == chs[i]:
            result.pop()
        else:
            result.append(chs[i])
        pre = result[-1]
    print(f"#{stage} {len(result)-1}")
```

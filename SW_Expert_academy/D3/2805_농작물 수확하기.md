# 내 코드
```py
T = int(input())
for stage in range(1, T+1):
    N = int(input())
    farm = [list(map(int, input())) for _ in range(N)]
    start_harvest = N//2 + 1
    end_harvest = N//2 - 1
    result = 0
    for row in range(N):
        if row <= N//2:
            start_harvest -= 1
            end_harvest += 1
        else:
            start_harvest += 1
            end_harvest -= 1
        result += sum(farm[row][start_harvest:end_harvest+1])
    print(f"#{stage} {result}")
```

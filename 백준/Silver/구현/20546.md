# 내 코드
```py
money = int(input())
stock = list(map(int, input().split()))
jun = [money, 0]
sung = [money, 0]
pre_stock = stock[0]
up = 0
down = 0
for today in stock[0:]:
    # 준현
    if today <= jun[0]:
        jun[1] += jun[0] // today
        jun[0] -= jun[1] * today
    # 성민
    if pre_stock < today:
        up += 1
        down = 0
    elif pre_stock > today:
        down += 1
        up = 0
    else:
        up = 0
        down = 0
    # 살지 말지
    if down == 3:
        if today <= sung[0]:
            sung[1] += sung[0] // today
            sung[0] -= sung[1] * today
    elif up == 3:
        if sung[1] != 0:
            sung[0] += sung[1] * today
            sung[1] = 0
    pre_stock = today

# 다 끝
sung_final = sung[0] + sung[1] * stock[-1]
jun_final = jun[0] + jun[1] * stock[-1]
if sung_final > jun_final:
    print("TIMING")
elif sung_final < jun_final:
    print("BNP")
else:
    print("SAMESAME")
```
# GPT 피드백
## 1. 가장 큰 버그 ① : 매수할 때 기존 보유 주식까지 곱해버림
문제점은 이거야:
- jun[1] += jun[0] // today 에서 오늘 새로 사는 주식 수를 jun[1]에 더하고
- 바로 아래에서 jun[0] -= jun[1] * today 로 전체 보유 주식 수 × 오늘 가격을 빼버리고 있어.

즉, 원래 갖고 있던 주식까지 포함해서 돈을 빼는 꼴이 됨 → 현금이 이상하게 마이너스나 줄어들어.

```py
if today <= jun_cash:
    cnt = jun_cash // today      # 오늘 새로 사는 주식 수
    jun_stock += cnt
    jun_cash -= cnt * today      # 새로 산 주식 수만큼만 돈 차감
```
=> 이렇게 고쳐야함.

## 2. 가장 큰 버그 ② : down == 3, up == 3 조건
- 여기서 중요한 건 “3일 연속”이 슬라이딩 윈도우라는 점이야.
- 즉, “3일 연속 하락 패턴”이 여러 번 겹쳐서 나올 수 있음 →
단순히 down == 3일 때만 매수하면 안 되고, down >= 3일 때마다 매수해야 해.

```py
if down >= 3:
    if today <= sung_cash:
        cnt = sung_cash // today
        sung_stock += cnt
        sung_cash -= cnt * today
elif up >= 3:
    if sung_stock > 0:
        sung_cash += sung_stock * today
        sung_stock = 0
```

## 3. 전체 수정 예시 코드
```py
money = int(input())
prices = list(map(int, input().split()))

# 준현: 현금, 주식 수
jun_cash, jun_stock = money, 0
# 성민: 현금, 주식 수
sung_cash, sung_stock = money, 0

up = down = 0
prev = prices[0]

for price in prices:
    # --- 준현 (BNP) ---
    if price <= jun_cash:
        cnt = jun_cash // price        # 오늘 새로 사는 주식 수
        jun_stock += cnt
        jun_cash -= cnt * price

    # --- 성민 (TIMING) ---
    # 추세 계산
    if price > prev:
        up += 1
        down = 0
    elif price < prev:
        down += 1
        up = 0
    else:
        up = down = 0

    # 3일 이상 하락 → 즉시 전량 매수
    if down >= 3:
        if price <= sung_cash:
            cnt = sung_cash // price
            sung_stock += cnt
            sung_cash -= cnt * price
    # 3일 이상 상승 → 즉시 전량 매도
    elif up >= 3:
        if sung_stock > 0:
            sung_cash += sung_stock * price
            sung_stock = 0

    prev = price

# 최종 자산 계산 (14일 종가 기준)
jun_total = jun_cash + jun_stock * prices[-1]
sung_total = sung_cash + sung_stock * prices[-1]

if jun_total > sung_total:
    print("BNP")
elif jun_total < sung_total:
    print("TIMING")
else:
    print("SAMESAME")
```

# 내 코드
```py
from collections import Counter
N = int(input())
result = Counter()
max_num = 0
for _ in range(N):
    book = input()
    result[book] += 1
    if result[book] > max_num:
        max_num = result[book]
result = sorted(result.items(), key=lambda x: (-int(x[1]), x[0]))
print(result[0][0])
```

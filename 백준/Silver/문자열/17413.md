# 내코드
```py
import sys
# 목적: 입력 S(10^5) -> 출력 단어만 뒤집
# 문자열 S: 알파벳 소문자 + 숫자 + 공백 + 특수문 / 문자열의 시작과 끝은 공백X / <과 >는 번갈아 등장. <이 먼저 등장(짝꿍으로 등장) / 태그는 <3이상의 부분문자열(알파벳 + 공백)> / 단어는 알파벳 + 숫자/ 단어는 공백으로 구분 / 태그는 단어X 

# 태그 만나면 아무것도X -> 태그 끝나면 stack에 저장 -> 태그 만나고 & stack 있으면 결과에 저장
input = sys.stdin.readline
tag = False
stack = []
result = ""
sen = input().strip() + " "
for ch in sen:
    if ch == "<":
        tag = True
        if stack:
            while stack:
                result += stack.pop()
        result += ch
        continue
    elif ch == ">":
        tag = False
        result += ch
        continue
    elif ch == " ":
        if stack:
            while stack:
                result += stack.pop()
        result += ch    
        continue    
    
    if not tag:
        stack.append(ch)
    else:
        result += ch
print(result)
```
- 시간 복잡도: O(N)
- 알고리즘 분류: 스택, 문자열

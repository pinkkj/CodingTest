# 내 코드
```py
s, t = input().split()
find = False
for i in range(0, len(t) - len(s) + 1):
    if t[i:i+len(s)] == s:
        print("Yes")
        find = True
        break

if not find:
    print("No")
```

# 개선
```py
import sys
for line in sys.stdin:
    line = line.strip()
    if not line:
        continue
    s, t = line.split()

    find = 0
    for ch in t:
        if ch == s[find]:
            find += 1
            if find == len(s):
                print("Yes")
                break
    
    if find != len(s):
        print("No")
```

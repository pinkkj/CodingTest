```py
from collections import Counter
# 한번 입었던 조합은 다신 안입음(꼭 하나는 바꿔야해)
# 주어진 의상들로 며칠 입을 수 있을까
T = int(input())
for _ in range(T):
    have = Counter()
    N = int(input())
    result = 1
    for _ in range(N):
        name, kind = input().split()
        have[kind] += 1
    for key, val in have.items():
        result *= (val+1)
    print(result - 1)
```

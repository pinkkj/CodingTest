# 내 코드
```py
# 필요하다고 생각하는 N개의 물건(무게 W/가치 V)
# 최대 K까지 들 수 있음 
N ,K = map(int, input().split())
info = [list(map(int, input().split())) for _ in range(N)]
info = sorted(info, key = lambda x: x[0])
dp = [0] * (K+1)
result = 0

for n in range(N): 
    for k in range(K, info[n][0]-1, -1):
        dp[k] = max(dp[k], dp[k-info[n][0]] + info[n][1])
print(max(dp))
```
- O(N*K)

```py
N, K = map(int, input().split())
items = [tuple(map(int, input().split())) for _ in range(N)]  # (W, V)

dp = [0] * (K + 1)

for w, v in items:
    for cap in range(K, w - 1, -1):  # cap = 현재 배낭 무게 한도
        dp[cap] = max(dp[cap], dp[cap - w] + v)

print(dp[K])
```

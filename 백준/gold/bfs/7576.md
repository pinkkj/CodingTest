```py
from collections import deque
M, N = map(int, input().split())
box = []
for _ in range(N):
    box.append(list(map(int, input().split())))
unrip = 0
result = -1
q = deque()
for i in range(N):
    for j in range(M):
        if box[i][j] == 0:
            unrip += 1
        elif box[i][j] == 1:
            q.append((i,j))
dir = [(1,0), (-1,0), (0,1), (0,-1)]
while q:
    for _ in range(len(q)):
        now_n, now_m = q.popleft()
        for move_m, move_n in dir:
            next_m = now_m + move_m
            next_n = now_n + move_n
            if (0 <= next_m < M) and (0 <= next_n < N):
                if box[next_n][next_m] == 0:
                    box[next_n][next_m] = 1
                    unrip -= 1
                    q.append((next_n, next_m))
    result += 1

if unrip != 0:
    result = -1
print(result)
```

def solution(fees, records):
    # 2 -> in / 3 -> out
    check = [] # in 이면 추가하고, out 이면 삭제하기
    answer = []
    time_sum = {}
    for record in records:
        time, number, how = record.split(" ")
        hour, minute = time.split(":")
        hour = int(hour)
        minute = int(minute)
        minute += hour * 60
        if number in check:
            time_sum[number] += minute
            check.remove(number)
        else:
            if number in list(time_sum.keys()):
                time_sum[number] -= minute
            else:
                time_sum[number] = -minute 
            check.append(number)  
    
    if len(check) != 0:
        for i in check:
            time_sum[i] += (60*23 + 59)
    
    time_sum = sorted(time_sum.items())
    print(time_sum)
    # // -> 몫 정수 / % -> 나머지 / / -> 실수 몫
    for record in time_sum:
        time = record[1]
        fee = fees[1]
        rest_time = time - fees[0]
        if rest_time > 0:
            if (rest_time % fees[2]) != 0:
                fee += ((rest_time//fees[2]) + 1) * fees[3]
            else:
                fee += ((rest_time//fees[2])) * fees[3]
        answer.append(fee)
        
    return answer

from collections import defaultdict
# defaultdict: defaultdict는 collections 모듈에 있는 딕셔너리의 하위 클래스로, 없는 키를 조회했을 때 자동으로 기본값을 만들어 넣어 주는 딕셔너리
from math import ceil # 올림함수

class Parking:
    def __init__(self, fees):
        self.fees = fees
        self.in_flag = False
        self.in_time = 0
        self.total = 0

    def update(self, t, inout):
        self.in_flag = True if inout=='IN' else False
        if self.in_flag:  self.in_time = str2int(t)
        else:             self.total  += (str2int(t)-self.in_time)

    def calc_fee(self):
        if self.in_flag: self.update('23:59', 'out')
        add_t = self.total - self.fees[0]
        return self.fees[1] + ceil(add_t/self.fees[2]) * self.fees[3] if add_t >= 0 else self.fees[1]

def str2int(string):
    return int(string[:2])*60 + int(string[3:])

def solution(fees, records):
    recordsDict = defaultdict(lambda:Parking(fees)) 
    for rcd in records:
        t, car, inout = rcd.split()
        recordsDict[car].update(t, inout)
        # 여기서 key가 없으면, 새로운 class 생성.
    return [v.calc_fee() for k, v in sorted(recordsDict.items())]

# 내 풀이
```py
# 목적: 불량 이용자 신고 / 처리 결과를 메일로 발송
# 조건 1. 각 유저 -> 한번에 한 명만! (횟수엔 제한X 한 유저에 대한 여러 신고->1)
# 조건 2. k번 이상 신고된 유저 -> 정지 / 해당 유저를 신고한 모든 유저에게 정지 메일
def solution(id_list, report, k):
    answer = {}
    target = []
    # 중복 제거
    report = set(report)
    report = list(report)
    # 누가 몇번 신고 당했나
    summary = {}
    for names in report:
        A, B = names.split(" ")
        if B in list(summary.keys()):
            summary[B][0] += 1
            summary[B][1].append(A)
        else:
            summary[B] = [1, [A]]
            
    for i in list(summary.keys()):
        if summary[i][0] >= k:
            target.extend(summary[i][1])
            
    for name in id_list:
        answer[name] = target.count(name)
    answer = list(answer.values())
        
    return answer
```

    dic_report = {id: [] for id in id_list} # 해당 유저를 신고한 ID
    for i in set(report):
        i = i.split()
        dic_report[i[1]].append(i[0])

    for key, value in dic_report.items():
        if len(value) >= k:
            for j in value:
                answer[id_list.index(j)] += 1

```py
def solution(friends, gifts):
    # 두 사람이 선물 주고받은 기록(더 많이 준 사람) -> 더 많이 준 사람이 받는다.
    # 둘다 똑같이 주거나 주고 받지X -> 선물 지수가 더 큰 사람이 받는다(같으면 아무도 받지X)
    # 선물 지수 = (준 선물) - (받은 선물)
    # 목표: 선물을 가장 많이 받을 친구가 받을 선물의 수!
    saved_dict = {}
    max_present = 0
    for name in friends:
        saved_dict[name] = {'give_to':[],'receive':[], 'score':0}
    for info in gifts:
        giver, receiver = info.split(" ")
        saved_dict[giver]['give_to'].append(receiver)
        saved_dict[giver]['score'] += 1
        saved_dict[receiver]['receive'].append(giver)
        saved_dict[receiver]['score'] -= 1

    for me in friends:
        present = 0
        for friend in friends:
            if (saved_dict[me]['give_to'].count(friend)) > (saved_dict[me]['receive'].count(friend)):
                present += 1
            elif (saved_dict[me]['give_to'].count(friend)) == (saved_dict[me]['receive'].count(friend)):
                if saved_dict[me]['score'] > saved_dict[friend]['score']:
                    present += 1
        max_present = max(max_present, present)

    return max_present
```

```py
def solution(friends, gifts):
    n = len(friends)
    idx = {name: i for i, name in enumerate(friends)}

    # pairwise 선물 횟수 테이블
    cnt = [[0]*n for _ in range(n)]
    # 선물 지수: 준 선물 - 받은 선물
    score = [0]*n

    # 기록 집계
    for g in gifts:
        a, b = g.split()
        ia, ib = idx[a], idx[b]
        cnt[ia][ib] += 1
        score[ia] += 1
        score[ib] -= 1

    # 다음 달 받을 선물 개수 세기
    receive_next = [0]*n
    for i in range(n):
        for j in range(i+1, n):  # (i, j) 페어 한 번만 판정
            if cnt[i][j] > cnt[j][i]:
                receive_next[i] += 1
            elif cnt[i][j] < cnt[j][i]:
                receive_next[j] += 1
            else:
                # 동수면 선물 지수 비교
                if score[i] > score[j]:
                    receive_next[i] += 1
                elif score[i] < score[j]:
                    receive_next[j] += 1
                # 같으면 아무도 못 받음

    return max(receive_next)
```
